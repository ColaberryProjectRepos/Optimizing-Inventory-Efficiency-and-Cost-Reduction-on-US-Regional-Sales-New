Step Name: Create DAX measures 15
Steps:
  1. Right click on the US Regional sales Table under the Data pane 
  2. From dropdown menu click on new measure 
  3. The formula bar appears along the top of the report canvas. 
  4.  Fill in all the DAX formulas provided below, one at a time, and when you finish Press Enter or select Commit (checkmark icon) in the formula bar.  
  5. After creating each measure choose the necessary format and decimal for each value (measure)
  6. Save
DAX Formulas for Calculated Measures and Calculated columns:

1. Inv_Level = 
 CALCULATE(
    SUM('US Regional Sales'[Order Quantity_x]),
    FILTER(
        'US Regional Sales',
        'US Regional Sales'[ProcuredDate]<=EARLIER('US Regiona Sales'[OrderDate])
    )
 )

 -
 CALCULATE(
    SUM('US Regional Sales'[Order Quantity_x]),
    FILTER(
        'US Regional Sales',
        'US Regional Sales'[ShipDate] <=EARLIER('US Regional Sales'[OrderDate])
    )
 )
2.Total Sales = SUMX('US Regional Sales ','US Regional Sales'[Unit Price] * 'US Regional Sales'[Order Quantity_x])

3.Total Discount = SUMX('US Regional Sales', 'US Regional Sales'[Unit Cost]* 'US Regional Sales'[Discount Applied_x])
4. Total Profit = SUMX('US Regional Sales', ('US Regional Sales'[Unit Price] - 'US Regional Sales '[Unit Cost]) * 'US Regional Sales'[Order Quantity_x])- [Total Discount]

5.TotalOrders = COUNTROWS('US Regional Sales')

6.Total Cost = SUMX('US Regional Sales', 'US Regional Sales'[Unit Cost] * 'US Regional Sales'[Order Quantity_x])

7.Cost per Order = DIVIDE([Total Cost], [TotalOrders])

8.Lead Time = DATEDIFF('US Regional Sales' [OrderDate], 'US Regional Sales'[DeliveryDate], DAY)

Average Unit Cost = DIVIDE(SUMX('US Regional Sales', 'US Regional Sales '[Unit Cost] * 'US Regional Sales'[Order Quantity_x]), SUM('US Regional Sales '[Order Quantity_x]))


9.Average Inventory Cost = Average('US Regional Sales'[Inv_Level])*[Average Unit Cost]

10.Average Lead Time = AVERAGEX('US Regional Sales', 'US Regional Sales'[Lead Time])
11. Lead Time% = [Average Lead Time]/ 100

12.GPM% = DIVIDE ([Total Profit], [Total Sales], 0)

13.Inventory Turnover Ratio = DIVIDE ([Total Sales], [Average Inventory Cost])

14.Inventory Turnover Ratio% = [Inventory Turnover Ratio]/ 100

15.InventoryTurnOverRatioArrow = IF([Inventory Turnover Ratio]<2,
            "?",  -- Down arrow for InventoryTurnoverRatio<2
            "?"   -- Up arrow for InventoryTurnoverRatio>2
        )

16.MonthsInInventory = DATEDIFF(
    MIN('US Regional Sales'[ProcuredDate]), 
    MAX('US Regional Sales'[OrderDate]), 
    Month
)

17.Total Sales MoM% = 
   DIVIDE(
        [Total Sales] - 
        CALCULATE([Total Sales], DATEADD('Calendar Year'[Date], -30, DAY)),
        CALCULATE([Total Sales], DATEADD('Calendar Year'[Date], -30, DAY))
    ) 


18.TotalSalesMoM%_Arrow = 
VAR Var_TotalSales_Change =
   DIVIDE(
        [Total Sales] - 
        CALCULATE([Total Sales], DATEADD('Calendar Year'[Date], -30, DAY)),
        CALCULATE([Total Sales], DATEADD('Calendar Year'[Date], -30, DAY))
    ) 
RETURN
    IF(
        ISBLANK( Var_TotalSales_Change ),  -- Check if the value is blank
        "",  -- Return empty string if the value is blank
        IF(
            Var_TotalSales_Change < 0,
            "?",  -- Down arrow for negative % values
            "?"   -- Up arrow for positive % values
        )
    )

19.TotalOrdersMoM% = 
   DIVIDE(
        [TotalOrders] - 
        CALCULATE([TotalOrders], DATEADD('Calendar Year'[Date], -30, DAY)),
        CALCULATE([TotalOrders], DATEADD('Calendar Year'[Date], -30, DAY))
    ) 

20.TotalOrdersMoM%_Arrow = 
VAR Var_TotalOrders_Change =
   DIVIDE(
        [TotalOrders] - 
        CALCULATE([TotalOrders], DATEADD('Calendar Year'[Date], -30, DAY)),
        CALCULATE([TotalOrders], DATEADD('Calendar Year'[Date], -30, DAY))
    ) 
RETURN
    IF(
        ISBLANK( Var_TotalOrders_Change ),  -- Check if the value is blank
        "",  -- Return empty string if the value is blank
        IF(
            Var_TotalOrders_Change < 0,
            "?",  -- Down arrow for negative % values
            "?"   -- Up arrow for positive % values
        )
    )
    
22.Stockout Event = IF('US Regional Sales '[Order Quantity_x] = 0, 1, 0)

23.Stockout Rate = DIVIDE(SUM('US Regional Sales'[Stockout Event]), COUNTROWS('US Regional Sales'))

24.ProcuredQTY = 
 CALCULATE(
    SUM('US Regional Sales'[Order Quantity_x]),
    FILTER(
        'US Regional Sales',
        'US Regional Sales'[ProcuredDate]<=EARLIER('US Regional Sales'[OrderDate])
    )
 )
25. ShippedQTY = CALCULATE(
    SUM('US Regional Sales'[Order Quantity_x]),
    FILTER(
        'US Regional Sales',
        'US Regional Sales'[ShipDate] <=EARLIER('US Regional Sales'[OrderDate])
    )
 )
26.TotalStockLeft = 
sum('US Regional Sales'[ProcuredQTY]) - Sum('US Regional Sales'[ShippedQTY])
